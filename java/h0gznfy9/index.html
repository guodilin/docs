    private static void forgetPassword(ArrayList<User> list) {
        Scanner s = new Scanner(System.in);

    private static void register(ArrayList<User> list) {

        Scanner s = new Scanner(System.in);

    private static boolean checkPhoneNumber(String phoneNumber) {
        // 手机号长度判断
        if (phoneNumber.length() != 11) {
            // 如果长度不等于18为 返回 false
            return false;
        }
        // 校验第一位不能为 0
        if (phoneNumber.startsWith("0")) {
            // 第一位如果是0 返回 false
            return false;
        }
        // 校验号码为数字
        // 校验 11 位数字是否满足
        for (int i = 0; i < phoneNumber.length(); i++) {
            // 获取前11位字符
            char c = phoneNumber.charAt(i);
            if (!(c >= '0' && c <= '9')) {
                return false;
            }
        }
        return true;

    }

    // 身份证校验
    private static boolean getIndex(String personID) {
        if (personID.length() != 18) {
            // 如果长度不等于18为 返回 false
            return false;
        }
        // 校验第一位不能为 0
        if (personID.startsWith("0")) {
            // 第一位如果是0 返回 false
            return false;
        }
        // 校验 17 位数字是否满足
        for (int i = 0; i < personID.length() - 1; i++) {
            // 获取前17位字符
            char c = personID.charAt(i);
            if (!(c >= '0' && c <= '9')) {
                return false;
            }
        }
        // 校验最后一位是字母 X和x 数字
        // 获取最后一位字符
        char endChar = personID.charAt(personID.length() - 1);
        // 判断最后一位字符是否满足要求
        return endChar == 'X' || endChar == 'x' || (endChar >= '0' && endChar <= '9');
    }

    // 判断用户是否存在
    private static boolean contains(ArrayList<User> list, String username) {
        for (int i = 0; i < list.size(); i++) {
            String listUsername = list.get(i).getUsername();
            if (listUsername.equals(username)) {
                return true;  // 重复返回 true
            }
        }
        // 当循环结束了，如果还没有即返回 false
        return false;
    }

    // checkUsername() 方法判断用户名是否符合条件 满足返回  true  不满足返回 false
    private static boolean checkUsername(String username) {
        int len = username.length();
        // 判断长度
        if (len < 3 || len > 15) {
            // 长度小于或者大于15位返回 false
            return false;
        }
        // 输入的类型必须是字母和数字
        for (int i = 0; i < username.length(); i++) {
            char c = username.charAt(i);
            // 取反，只要不包含字母数字即返回 false
            if (!((c >= 'a' && c <= 'z') || (c >= 'A' && c <= 'Z') || (c >= '0' && c <= '9'))) {
                return false; //遇到即 return 结束循环。
            }
        }
        // 不能是纯数字
        int count = 0;
        for (int i = 0; i < username.length(); i++) {
            char c = username.charAt(i);
            // if (c >= '0' && c <= '9') {   // 方法1 存在数字 计数一次，不能等于username长度，相等是纯数字
            // 方法2 反像理解
            if ((c >= 'a' && c <= 'z') || (c >= 'A' && c <= 'Z')) {
                // 只要包含了其中一位字母，计数一次
                count++;
                // 只要包含了其中一位字母，即满足结束循环，提高性能
                break;
            }
        }
        // return count < len;
        // 大于0即有字母存在，即不存在纯数字
        return count > 0;
    }

    // 登录验证
    private static void login(ArrayList<User> list) {
        System.out.println("登录");

添加手机对象并返回需求的数据

需求

定义Javabean类：Phone

Phone属性：品牌，价格。

main方法中定义一个集合，存入三个手机对象。

分别为：小米，1000。苹果，8000。锤子，2999。

定义一个方法，将价格低于3000的手机信息返回。

import java.util.ArrayList;

public class Dome {
    public static void main(String[] args) {
        // 定义集合
        ArrayList<Phone> list = new ArrayList<>();
        // 定义对象并存入内容
        Phone p1 = new Phone("小米", 1000);
        Phone p2 = new Phone("苹果", 8000);
        Phone p3 = new Phone("催子", 2999);
        // 把对象存入集合
        list.add(p1);
        list.add(p2);
        list.add(p3);
        // 调用返回小于3000的商品
        ArrayList<Phone> price = getPrice(list);
        // 循环遍历集合
        for (int i = 0; i < price.size(); i++) {
            String str = price.get(i).toString();
            System.out.println(str);
        }
    }

    public static ArrayList<Phone> getPrice(ArrayList<Phone> list) {
        // 如果有多个小于3000价格的想要返回，则直接新建一个集合
        ArrayList<Phone> newList = new ArrayList<>();
        for (int i = 0; i < list.size(); i++) {
            // 判断小于3000价格的
            if (list.get(i).getPrice() < 3000) {
                // 如果小于3000价格的添加至新集合
                newList.add(list.get(i));
            }
        }
        // 循环结束后返回新集合
        return newList;
    }
}
